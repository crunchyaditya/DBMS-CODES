---------------------------------------------------
-- Create Function
---------------------------------------------------

CREATE FUNCTION Get_Grade(marks INT)
RETURNS VARCHAR(20)

DETERMINISTIC

BEGIN

DECLARE grade VARCHAR(20);

IF marks >= 80 THEN
    SET grade = 'Distinction';

ELSEIF marks >= 60 THEN
    SET grade = 'First Class';

ELSEIF marks >= 40 THEN
    SET grade = 'Pass';

ELSE
    SET grade = 'Fail';

END IF;

RETURN grade;

END //

---------------------------------------------------
-- Execute Function
---------------------------------------------------

SELECT Student_Name,
Marks,
Get_Grade(Marks) AS Grade
FROM Student;

-- Create Database
CREATE DATABASE College_DB;

-- Select Database
USE College_DB;

-- Create Student Table
CREATE TABLE Student (
    Student_ID INT PRIMARY KEY,
    Student_Name VARCHAR(50),
    Age INT,
    City VARCHAR(50),
    Marks INT
);

---------------------------------------------------
-- 1. Names Starting With 'R'
---------------------------------------------------

SELECT * FROM Student
WHERE Student_Name LIKE 'R%';

---------------------------------------------------
-- 2. Names Ending With 'a'
---------------------------------------------------

SELECT * FROM Student
WHERE Student_Name LIKE '%a';

---------------------------------------------------
-- 3. Names Containing 'h'
---------------------------------------------------

SELECT * FROM Student
WHERE Student_Name LIKE '%h%';

---------------------------------------------------
-- 4. City Starts With 'P'
---------------------------------------------------

SELECT * FROM Student
WHERE City LIKE 'P%';

---------------------------------------------------
-- 5. Exactly 5-Letter Names Starting With 'R'
---------------------------------------------------

SELECT * FROM Student
WHERE Student_Name LIKE 'R____';

---------------------------------------------------
-- 6. Names Having 'i' in Second Position
---------------------------------------------------

SELECT * FROM Student
WHERE Student_Name LIKE '_i%';

-- Create Database
CREATE DATABASE School_DB;

-- Select Database
USE School_DB;

-- Create Student Table
CREATE TABLE Student (
    Student_ID INT PRIMARY KEY,
    Student_Name VARCHAR(50),
    Age INT,
    City VARCHAR(50)
);

-- Insert Records
INSERT INTO Student
VALUES

-- Display Table
SELECT * FROM Student;

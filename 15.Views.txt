-- Create Database
CREATE DATABASE College_DB;

-- Select Database
USE College_DB;

---------------------------------------------------
-- Create Student Table
---------------------------------------------------

CREATE TABLE Student (
    Student_ID INT PRIMARY KEY,
    Student_Name VARCHAR(50),
    City VARCHAR(50),
    Marks INT
);

SELECT * FROM Student;

---------------------------------------------------
-- Create View
---------------------------------------------------

CREATE VIEW Student_View AS
SELECT Student_Name, Marks
FROM Student;

---------------------------------------------------
-- Display View
---------------------------------------------------

SELECT * FROM Student_View;

---------------------------------------------------
-- Create Another View
---------------------------------------------------

CREATE VIEW Pune_Students AS
SELECT *
FROM Student
WHERE City = 'Pune';

---------------------------------------------------
-- Display Pune Students View
---------------------------------------------------

SELECT * FROM Pune_Students;

---------------------------------------------------
-- Update View
---------------------------------------------------

CREATE OR REPLACE VIEW Student_View AS
SELECT Student_Name, City, Marks
FROM Student;

---------------------------------------------------
-- Display Updated View
---------------------------------------------------

SELECT * FROM Student_View;

---------------------------------------------------
-- Drop View
---------------------------------------------------

DROP VIEW Student_View;

-- Show Existing Tables and Views
---------------------------------------------------

-- Create Database
CREATE DATABASE College_DB;

-- Select Database
USE College_DB;

---------------------------------------------------
-- Create Student Table
---------------------------------------------------

CREATE TABLE Student (
    Student_ID INT PRIMARY KEY,
    Student_Name VARCHAR(50),
    City VARCHAR(50),
    Marks INT,
    Age INT
);

---------------------------------------------------
-- 1. Sort Marks in Ascending Order
---------------------------------------------------

SELECT * FROM Student
ORDER BY Marks ASC;

---------------------------------------------------
-- 2. Sort Marks in Descending Order
---------------------------------------------------

SELECT * FROM Student
ORDER BY Marks DESC;

---------------------------------------------------
-- 3. Sort Student Names Alphabetically
---------------------------------------------------

SELECT * FROM Student
ORDER BY Student_Name ASC;

---------------------------------------------------
-- 4. Sort According to City Name
---------------------------------------------------

SELECT * FROM Student
ORDER BY City ASC;

---------------------------------------------------
-- 5. Sort by Age (Highest to Lowest)
---------------------------------------------------

SELECT * FROM Student
ORDER BY Age DESC;

---------------------------------------------------
-- 6. Multiple Sorting
---------------------------------------------------

SELECT * FROM Student
ORDER BY City ASC, Marks DESC;
